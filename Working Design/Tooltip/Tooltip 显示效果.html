  1. 使用纯HTML/CSS/JS实现Tooltip功能
  2. 展示正确的Angular代码结构

代码示例

以下是在HTML模板中使用matTooltip的代码：

<div *ngIf="!this.fieldsMap.username.show"> <input type="checkbox" id="isTechnicalUser" name="isTechUser" style="margin-left: 10px !important; vertical-align: middle;"> <label class="field-label ng-tns-c1-9 ng-star-inserted" for="isTechnicalUser" style="font-size: 13px; margin-left: 5px;"> {{'administration.users.wum.isTechUser' | translate}} </label> <i class="fa-lg fa fa-info-circle message-info" matTooltip="{{'administration.users.wum.isTechUser.info' | translate}}" matTooltipPosition="right" style="margin-left: 10px; cursor: pointer;"></i> </div>

import { Component } from '@angular/core'; import { MatTooltipModule } from '@angular/material/tooltip'; @Component({ selector: 'app-user-form', templateUrl: './user-form.component.html', styleUrls: ['./user-form.component.css'] }) export class UserFormComponent { // 组件逻辑 }
